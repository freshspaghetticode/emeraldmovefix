function appendElementIf({id, wAbove, wBelow, hAbove, hBelow}) {

	// find browser height
	var w = window.innerWidth;
	var h = window.innerHeight;

	// If true after browser size check then unit is added

	// Append element if condition is met
	if (condition) {
		createAndAppendElement(id);
	}
}
