Smeargle can learn any set of moves due to <a href="/wiki/Sketch_(move)" title="Sketch (move)">Sketch</a>, which permanently copies any move. However, Sketch cannot be used to learn Sketch, <a href="/wiki/Struggle_(move)" title="Struggle (move)">Struggle</a> or <a href="/wiki/Shadow_move" title="Shadow move">Shadow moves</a>.

function appendElementIf({id, wAbove, wBelow, hAbove, hBelow}) {

	// find browser height
	var w = window.innerWidth;

	// If true after browser size check then unit is added

	// Append element if condition is met
	if (condition) {
		createAndAppendElement(id);
	}
